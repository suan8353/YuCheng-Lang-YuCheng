pub struct HintProvider {
    user_profile: UserProfile,
    hint_rules: Vec<HintRule>,

impl HintProvider {
    pub fn get_hints(&self, document: &Document, position: Position) -> Vec<Hint> {
        let mut hints = Vec::new();
        
        for rule in &self.hint_rules {
            if rule.should_trigger(document, position, &self.user_profile) {
                hints.push(rule.generate_hint(document, position));
            }
        }
        
        hints
    }
}
